class Husdjur {
    constructor(djurnamn) {
        this.namn = djurnamn;
        this.glad = false;
        this.hungrig = 4;
    }

    visa() {
        console.log(this.namn)
        if (this.glad) {
            console.log("är glad!");
        }
        else {
            console.log("är ledsen");
        }
        if (this.hungrig > 0) {
            console.log("och hungrig...");
        }
    }
}
